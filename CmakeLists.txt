# CMakeLists files in this project can
# refer to the root source directory of the project as ${ROMAN_SOURCE_DIR} and
# to the root binary directory of the project as ${ROMAN_BINARY_DIR}.
cmake_minimum_required (VERSION 2.8.11)
project (ROMAN)

# Recurse into the "src" and "tests" subdirectories. This does not actually
# cause another cmake executable to run. The same process will walk through
# the project's entire directory structure.
add_subdirectory (src)
add_subdirectory (tests)
